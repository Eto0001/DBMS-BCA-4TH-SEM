    Id INT PRIMARY KEY,
    Tname VARCHAR(20)
);

CREATE TABLE Student (
    Id INT PRIMARY KEY,
    Sname VARCHAR(20)
);

SELECT Tname AS Name FROM Teacher
UNION
SELECT Sname FROM Student;


SELECT DISTINCT T.Tname AS Common_Name
FROM Teacher T
INNER JOIN Student S
ON T.Tname = S.Sname;


SELECT DISTINCT Tname AS Common_Name
FROM Teacher
WHERE Tname IN (SELECT Sname FROM Student);


SELECT DISTINCT T.Tname AS Only_Teacher
FROM Teacher T
LEFT JOIN Student S
ON T.Tname = S.Sname
WHERE S.Sname IS NULL;
