CREATE VIEW Emp_stat AS
SELECT Marital_status AS Status, SUM(Salary) AS Total_Salary
FROM Employee
GROUP BY Marital_status;

SELECT * FROM Emp_stat;

SET Total_Salary = 50000
WHERE Status = 'Married';
